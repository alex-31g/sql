Таблица **user** предназначена для хранения пользователей.  
Пользователь имеет возможность подписываться на других пользователей, список которых хранится в таблице **follower**.  
Между данными таблицами существует связь 'один-ко-многим', это значит что: один пользователь может иметь много подписок.

```sql
CREATE DATABASE social_network;
SHOW DATABASES;
USE social_network;

CREATE TABLE user (
	id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
	name VARCHAR(30) NOT NULL
);
CREATE TABLE follower (
	id_user INT UNSIGNED,
	id_following INT,
	FOREIGN KEY (id_user) REFERENCES user(id)
);
SHOW TABLES;
SHOW COLUMNS FROM user;
SHOW COLUMNS FROM follower;

# Добавить новый столбец age в таблицу user:
ALTER TABLE user ADD age INT;

# Добавить нового пользователя в таблицу user:
INSERT INTO user VALUES (NULL, 'tayson', 40);

# Пользователь с id=3 подписывается на пользователя с id=8:
INSERT INTO follower VALUES (3, 8);

# На какое количество человек подписан пользователь с id=3:
SELECT * FROM follower WHERE id_user=3;

# Сколько человек подписано на пользователя с id=3:
SELECT * FROM follower WHERE id_following=3;
```
